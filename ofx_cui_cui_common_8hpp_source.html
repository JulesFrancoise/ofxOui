148  static ofTrueTypeFont* requestFont(void* client, string font_path,
149  float font_size, float line_height,
150  float letter_spacing) {
151  pair<string, vector<float>> fontSettings(
152  font_path, {font_size, line_height, letter_spacing});
153  for (auto& f : getInstance().catalog_) {
154  if (f.second.second.count(client) > 0) {
155  if (f.first == fontSettings) return f.second.first;
156  f.second.second.erase(client);
157  if (f.second.second.empty()) {
158  delete f.second.first;
159  getInstance().catalog_.erase(f.first);
160  }
161  break;
162  }
163  }
164  auto it = getInstance().catalog_.find(fontSettings);
165  if (it == getInstance().catalog_.end()) {
166  ofTrueTypeFont* font = new ofTrueTypeFont();
167  font->load(font_path, getRetina() ? font_size * 2. : font_size);
168  font->setLineHeight(getRetina() ? line_height * 2. : line_height);
169  font->setLetterSpacing(letter_spacing);
170  getInstance().catalog_[fontSettings].first = font;
171  getInstance().catalog_.at(fontSettings).second.insert(client);
172  return font;
173  } else {
174  getInstance().catalog_.at(fontSettings).second.insert(client);
175  return it->second.first;
176  }
177  }
